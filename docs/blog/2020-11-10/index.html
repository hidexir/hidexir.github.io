調査内容

package main
import (
    "log"
)
var (
	id int
	name string
)
rows, err := db.Query("SELECT id, name FROM users WHERE id = ?", 1)
if err != nil {
	log.Fatal(err)
}
defer rows.Close()
for rows.Next() {
	err := rows.Scan(&id, &name)
	if err != nil {
		log.Fatal(err)
	}
	log.Println(id, name)
}
err = rows.Err()
if err != nil {
	log.Fatal(err)
}
